    const BOM = '﻿';

    function sanitizeCsvCell(value: string): string {
      // Mitigation T-05-07-04 + T-06-LIASSE-01
      const dangerousPrefixes = ['=', '+', '-', '@', '\t', '\r'];

      return value;
    }
